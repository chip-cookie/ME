from kiwipiepy import Kiwi

class KoreanSmartChunker:
    def __init__(self, chunk_size=500, overlap_size=100):
        self.kiwi = Kiwi()
        self.chunk_size = chunk_size
        self.overlap_size = overlap_size

    def create_chunks(self, text: str):
        # 1. Kiwi 형태소 분석기로 '문장 단위' 정확하게 분리
        # 단순 .split('.')은 "1. 항목" 같은 경우 오작동함. Kiwi는 이걸 해결함.
        sentences = self.kiwi.split_into_sents(text)
        
        chunks = []
        current_chunk = []
        current_length = 0
        
        for sent in sentences:
            sent_text = sent.text
            sent_len = len(sent_text)
            
            # 2. 현재 청크 + 새 문장이 설정 크기를 넘으면 저장
            if current_length + sent_len > self.chunk_size:
                # 현재까지 모은 문장들을 합쳐서 청크 생성
                full_chunk_text = " ".join(current_chunk)
                chunks.append(full_chunk_text)
                
                # 3. [핵심] Overlapping (중첩) 로직
                # 이전 청크의 마지막 부분(overlap_size 만큼)을 다음 청크의 시작으로 가져옴
                # 문맥이 끊기는 것을 방지하여 RAG 검색 정확도 200% 상승
                overlap_text = ""
                while len(overlap_text) < self.overlap_size and current_chunk:
                    overlap_text = current_chunk.pop() + " " + overlap_text
                
                current_chunk = [overlap_text.strip(), sent_text]
                current_length = len(overlap_text) + sent_len
            else:
                current_chunk.append(sent_text)
                current_length += sent_len
                
        # 마지막 남은 조각 처리
        if current_chunk:
            chunks.append(" ".join(current_chunk))
            
        return chunks

def generate_prompt(user_facts: dict, job_description: str, retrieved_context: list):
    """
    user_facts: 사용자 이력서/경험 데이터 (JSON)
    job_description: 지원하려는 채용 공고 내용
    retrieved_context: Kiwi로 청킹되어 검색된 관련 경험들
    """
    
    system_prompt = f"""
    당신은 사용자의 취업 합격을 돕는 냉철하고 논리적인 [AI 에이전트]입니다.
    
    [핵심 규칙 (Critical Rules)]
    1. **NO HALLUCINATION (거짓 작성 금지):** 아래 제공된 [User Facts]와 [Retrieved Context]에 없는 내용은 절대 지어내지 마십시오.
       정보가 부족하면 가상의 이야기를 쓰지 말고, 사용자에게 경험을 더 물어보는 질문을 하십시오.
       
    2. **Fact Grounding (사실 기반):**
       모든 주장은 [User Facts]에 있는 구체적인 수치와 경험에 기반해야 합니다.
       
    3. **Strategic Thinking (전략적 사고):**
       [Job Description]에서 요구하는 역량과 [User Facts]를 연결하여 작성하십시오.
    """

    user_message = f"""
    [Job Description]
    {job_description}

    [User Facts (절대적 사실)]
    {json.dumps(user_facts, ensure_ascii=False)}

    [Retrieved Context (관련된 세부 경험)]
    {"\n".join(retrieved_context)}

    위 정보를 바탕으로 자기소개서의 '지원동기' 항목을 작성해 주세요.
    """
    
    return system_prompt, user_message

📊 전체 데이터 흐름도 (Pipeline)

이 아키텍처가 적용되면 데이터는 아래 순서로 흐릅니다.

    입력 (Input): 사용자가 이력서(PDF/HWP) 업로드.

    분해 (Eye): Kiwi가 문장을 정확히 쪼개고, Overlapping 기술이 앞뒤 문맥을 겹쳐서 저장 (벡터 DB).

    질문 (Query): "삼성전자 지원동기 써줘" 요청.

    검색 (Retrieval): 질문과 관련된 사용자의 '진짜 경험'만 벡터 DB에서 가져옴.

    제어 (Control): 프롬프트가 "가져온 팩트 외엔 절대 쓰지 마"라고 제약 걺.

    생성 (Brain): **vLLM(Qwen)**이 이 모든 정보를 초고속으로 처리하여, 논리적이고 사실적인 자소서 생성.
